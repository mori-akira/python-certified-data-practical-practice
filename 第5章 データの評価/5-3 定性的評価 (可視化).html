df_avg_weight_year = (
    df.assign(Year=df.loc[:, "Date_Egg"].dt.to_period("Y"))
    .assign(Year=lambda e: e.loc[:, "Year"].dt.to_timestamp())
    .groupby(["Species_Short", "Year"], as_index=False, observed=True)["Body_Mass"]
    .mean()
)
